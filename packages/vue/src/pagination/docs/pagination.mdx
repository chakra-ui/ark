---
id: pagination
name: Pagination
description:
  Pagination is an interface that allows navigating between pages that contain
  split information, instead of being shown on a single page.
---

## Import

```ts
import {
  Pagination,
  PaginationEllipsis,
  PaginationList,
  PaginationListItem,
  PaginationNextPageTrigger,
  PaginationPageTrigger,
  PaginationPrevPageTrigger,
} from '@ark-ui/vue'
```

## Usage

Here's an example of how to use the Pagination component:

```vue
<script setup lang="ts">
import {
  Pagination,
  PaginationEllipsis,
  PaginationList,
  PaginationNextPageTrigger,
  PaginationPageTrigger,
  PaginationPrevPageTrigger,
  type PaginationContext,
} from '@ark-ui/vue'
</script>
<template>
  <Pagination
    :count="5000"
    :page-size="10"
    :sibling-count="2"
    v-slot="{ pages }: PaginationContext"
  >
    <PaginationList>
      <PaginationListItem>
        <PaginationPrevPageTrigger> Previous Page </PaginationPrevPageTrigger>
      </PaginationListItem>

      <template v-for="(page, index) in pages" :key="index">
        <PaginationListItem>
          <PaginationPageTrigger
            v-if="page.type === 'page'"
            :value="page.value"
            :type="page.type"
          >
            {{ page.value }}
          </PaginationPageTrigger>
          <PaginationEllipsis v-else :index="index">
            &#8230;
          </PaginationEllipsis>
        </PaginationListItem>
      </template>
      <PaginationListItem>
        <PaginationNextPageTrigger> Next Page </PaginationNextPageTrigger>
      </PaginationListItem>
    </PaginationList>
  </Pagination>
</template>
```

## Controlled Pagination

To create a controlled Pagination component, manage the state of the current
page using the `page` prop and update it when the `onChange` event handler is
called:

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { Pagination, type PaginationProps } from '@ark-ui/vue'

const currentPage = ref<PaginationProps['modelValue']>(1)
</script>
<template>
  <Pagination
    :count="5000"
    :page-size="10"
    :sibling-count="2"
    v-model="currentPage"
  >
    <!-- ... -->
  </Pagination>
</template>
```

## Customizing Pagination

You can customize the Pagination component by setting various props such as
`dir`, `pageSize`, `siblingCount`, and `translations`. Here's an example of a
customized Pagination:

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { Pagination } from '@ark-ui/vue'
</script>
<template>
  <Pagination
    :count="5000"
    :page-size="10"
    :sibling-count="3"
    dir="ltr"
    :translations="{
      nextPageTriggerLabel: 'Next',
      prevPageTriggerLabel: 'Prev',
      pageTriggerLabel: (details) => `Page ${details.page}`,
    }"
  >
    <!-- ... -->
  </Pagination>
</template>
```
