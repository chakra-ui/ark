---
id: accordion
name: Accordion
description:
  An accordion is a vertically stacked set of interactive headings containing a
  title, content snippet, or thumbnail representing a section of content.
---

## Import

```ts
import {
  Accordion,
  AccordionTrigger,
  AccordionItem,
  AccordionContent,
} from '@ark-ui/vue'
```

## Usage

The Accordion component consists of the `Accordion`, `AccordionTrigger`,
`AccordionItem` and `AccordionContent` components.

```vue
<script setup>
import {
  Accordion,
  AccordionTrigger,
  AccordionItem,
  AccordionContent,
} from '@ark-ui/vue'
</script>
<template>
  <Accordion>
    <AccordionItem value="panel-1">
      <AccordionTrigger>
        <button>Panel 1 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 1 content</AccordionContent>
    </AccordionItem>

    <AccordionItem value="panel-2">
      <AccordionTrigger>
        <button>Panel 2 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 2 content</AccordionContent>
    </AccordionItem>
  </Accordion>
</template>
```

### Allow toggling the open state

To collapse an already expanded accordion item by clicking on it, set the
`collapsible` prop property to `true`.

:::note

If `multiple` is `true`, we internally set collapsible to be true.

:::

```vue
<script setup>
import {
  Accordion,
  AccordionTrigger,
  AccordionItem,
  AccordionContent,
} from '@ark-ui/vue'
</script>
<template>
  <Accordion collapsible>
    <AccordionItem value="panel-1">
      <AccordionTrigger>
        <button>Panel 1 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 1 content</AccordionContent>
    </AccordionItem>

    <AccordionItem value="panel-2">
      <AccordionTrigger>
        <button>Panel 2 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 2 content</AccordionContent>
    </AccordionItem>
  </Accordion>
</template>
```

### Allow multiple open accordions at once

To allow multiple items to be expanded at once, set `multiple` to `true`. This
mode implicitly sets `collapsible` to `true` and ensures that each accordion can
be expanded.

```vue
<script setup>
import {
  Accordion,
  AccordionTrigger,
  AccordionItem,
  AccordionContent,
} from '@ark-ui/vue'
</script>
<template>
  <Accordion multiple>
    <AccordionItem value="panel-1">
      <AccordionTrigger>
        <button>Panel 1 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 1 content</AccordionContent>
    </AccordionItem>

    <AccordionItem value="panel-2">
      <AccordionTrigger>
        <button>Panel 2 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 2 content</AccordionContent>
    </AccordionItem>
  </Accordion>
</template>
```

### Disable specific accordion items

To disable a specific accordion item, set the `disabled` prop to `true`. When an
accordion item is disabled, it is skipped from keyboard navigation and can't be
interacted with.

```vue
<script setup>
import {
  Accordion,
  AccordionTrigger,
  AccordionItem,
  AccordionContent,
} from '@ark-ui/vue'
</script>
<template>
  <Accordion>
    <AccordionItem value="panel-1" disabled>
      <AccordionTrigger>
        <button>Panel 1 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 1 content</AccordionContent>
    </AccordionItem>

    <AccordionItem value="panel-2">
      <AccordionTrigger>
        <button>Panel 2 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 2 content</AccordionContent>
    </AccordionItem>
  </Accordion>
</template>
```

### Uncontrolled usage with `defaultValue`

Expand accordion items by default with setting `defaultValue` to the value of an
`AccordionItem`. The `defaultValue` should be of type `string | string[] | null`

```vue
<script setup>
import {
  Accordion,
  AccordionTrigger,
  AccordionItem,
  AccordionContent,
} from '@ark-ui/vue'
</script>
<template>
  <Accordion defaultValue="panel-2">
    <AccordionItem value="panel-1">
      <AccordionTrigger>
        <button>Panel 1 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 1 content</AccordionContent>
    </AccordionItem>

    <AccordionItem value="panel-2">
      <AccordionTrigger>
        <button>Panel 2 trigger</button>
      </AccordionTrigger>
      <AccordionContent>Panel 2 content</AccordionContent>
    </AccordionItem>
  </Accordion>
</template>
```

### Controlled usage with `v-model`

The accordion item accepts a `v-model` prop, to allow a controlled usage. The
`v-model` prop should be of type `string | string[] | null`.

```vue
<script setup lang="ts">
import { ref } from 'vue'
import {
  Accordion,
  AccordionTrigger,
  AccordionItem,
  AccordionContent,
} from '@ark-ui/vue'

const value = ref('panel-2')
</script>
<template>
  <Accordion v-model="value">
    <AccordionItem
      v-for="id in ['panel-1', 'panel-2', 'panel-3']"
      :key="item"
      :value="id"
    >
      <AccordionTrigger>
        <button>{{ id }} trigger</button>
      </AccordionTrigger>
      <AccordionContent>{{ id }} content</AccordionContent>
    </AccordionItem>
  </Accordion>
</template>
```
