---
id: pressable
name: Pressable
description:
  Pressable handles press interactions across mouse, touch, keyboard, and screen
  readers.
---

## Import

```ts
import { Pressable } from '@ark-ui/vue'
```

## Usage

The Pressable component consists only of the `Pressable` component.

```tsx
<script setup lang='ts'>
  import { Pressable } from '@ark-ui/vue'

  const handlePress = () => alert('onPress')
</script>
<template>
  <Pressable @press="handlePress">Press me</Pressable>
</template>
```

### Long Press

`onLongPress` is called when the element has been pressed for 500 milliseconds.

```tsx
<script setup lang='ts'>
  import { Pressable } from '@ark-ui/vue'

  const handleLongPress = () => alert('onPress')
</script>
<template>
  <Pressable @long-press="handlePress">Press me long</Pressable>
</template>
```

### Disabled

To disable a pressable, set the `disabled` prop to `true`.

```tsx
<Pressable isDisabled>Disabled</Pressable>
```
