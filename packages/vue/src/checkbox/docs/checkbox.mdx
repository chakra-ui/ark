---
id: checkbox
name: Checkbox
description:
  A checkbox allows users to make a binary choice, i.e. a choice between one of
  two possible mutually exclusive options.
---

## Import

```ts
import {
  Checkbox,
  CheckboxControl,
  CheckboxInput,
  CheckboxLabel,
} from '@ark-ui/vue'
```

## Usage

The Checkbox component consists of the `CheckboxControl`, `CheckboxInput`,
`CheckboxLabel` components.

### Plain

```vue
<script setup lang="ts">
import {
  Checkbox,
  CheckboxControl,
  CheckboxInput,
  CheckboxLabel,
} from '@ark-ui/vue'
</script>
<template>
  <Checkbox>
    <CheckboxLabel>Checkbox</CheckboxLabel>
    <CheckboxInput />
    <CheckboxControl />
  </Checkbox>
</template>
```

### Controlled usage

When the checkbox value changes, the `onChange` callback is invoked. Use the
`v-model` to handle the `checked` state

```vue
<script setup lang="ts">
import { ref } from 'vue'
import {
  Checkbox,
  CheckboxControl,
  CheckboxInput,
  CheckboxLabel,
} from '@ark-ui/vue'

const checked = ref(false)
</script>
<template>
  <Checkbox v-model="checked">
    <CheckboxLabel>Controlled</CheckboxLabel>
    <CheckboxInput />
    <CheckboxControl />
  </Checkbox>
</template>
```

### Indeterminate

To make a checkbox indeterminate, set the `indeterminate` prop to `true`

```vue
<script setup lang="ts">
import { reactive, ref, watch } from 'vue'
import {
  Checkbox,
  CheckboxControl,
  CheckboxInput,
  CheckboxLabel,
  type CheckboxProps,
} from '@ark-ui/vue'

const parentChecked = ref<CheckboxProps['modelValue']>(false)
const childCheckedItems = reactive<CheckboxProps['modelValue'][]>([
  false,
  false,
])

watch(childCheckedItems, (items) => {
  if (items.indexOf(true) < 0) {
    parentChecked.value = false
    return
  }
  parentChecked.value = items.every(Boolean) ? true : 'indeterminate'
})

watch(parentChecked, (parentVal) => {
  if (parentVal === 'indeterminate') return
  childCheckedItems.map((_, idx) => {
    childCheckedItems[idx] = parentVal
  })
  return
})
</script>

<template>
  <Checkbox v-model="parentChecked">
    <CheckboxControl />
    <CheckboxLabel> Parent Checkbox</CheckboxLabel>
    <CheckboxInput />
  </Checkbox>
  <div>
    <Checkbox v-model="childCheckedItems[0]">
      <CheckboxControl />
      <CheckboxLabel> Child One Checkbox</CheckboxLabel>
      <CheckboxInput />
    </Checkbox>
    <Checkbox v-model="childCheckedItems[1]">
      <CheckboxControl />
      <CheckboxLabel>Child Two Checkbox</CheckboxLabel>
      <CheckboxInput />
    </Checkbox>
  </div>
</template>
```

## V-Slot Usage

For cases where you need more flexibility in rendering, the Checkbox component
offers the use of the `v-slot`. This gives you access to the checkbox's API,
allowing you to customize the checkbox control's rendering:

```vue
<script setup lang="ts">
import {
  Checkbox,
  CheckboxControl,
  CheckboxInput,
  CheckboxLabel,
  type CheckboxContext,
} from '@ark-ui/vue'
</script>
<template>
  <Checkbox v-slot="{ isChecked }: CheckboxContext">
    <CheckboxLabel>Checkbox with slot props: {{ isChecked }}</CheckboxLabel>
    <CheckboxInput />
    <CheckboxControl />
  </Checkbox>
</template>
```

## Conclusion

The Checkbox component provides a robust and flexible solution for toggleable
input needs in your applications. With its support for controlled and
indeterminate states, as well as custom render props, it offers a wide range of
design possibilities and use cases.
