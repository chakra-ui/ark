---
id: pagination
name: Pagination
description:
  Pagination is an interface that allows navigating between pages that contain
  split information, instead of being shown on a single page.
---

## Import

```ts
import {
  Pagination,
  PaginationEllipsis,
  PaginationList,
  PaginationNextPageTrigger,
  PaginationPageTrigger,
  PaginationPrevPageTrigger,
} from '@ark-ui/react'
```

## Usage

The Pagination component consists of the `PaginationEllipsis`, `PaginationList`,
`PaginationNextPageTrigger`, `PaginationPageTrigger`, and
`PaginationPrevPageTrigger` components.

```tsx
<Pagination count={5000} pageSize={10} siblingCount={2}>
  {({ pages }) => (
    <PaginationList>
      <PaginationPrevPageTrigger>
        <button>Previous</button>
      </PaginationPrevPageTrigger>

      {pages.map((page, index) =>
        page.type === 'page' ? (
          <PaginationPageTrigger key={index} value={page.value}>
            <button>{page.value}</button>
          </PaginationPageTrigger>
        ) : (
          <PaginationEllipsis key={index} index={index}>
            &#8230;
          </PaginationEllipsis>
        ),
      )}

      <PaginationNextPageTrigger>
        <button>Next</button>
      </PaginationNextPageTrigger>
    </PaginationList>
  )}
</Pagination>
```
