```tsx
<DatePicker>
  {(api) => (
    <DatePickerContent>
      {api.view === 'day' && (
        <DatePickerGrid>
          <DatePickerRowHeader>
            <DatePickerColumnHeader />
          </DatePickerRowHeader>
          <DatePickerRowGroup>
            {api.weeks.map((week, id) => (
              <DatePickerRow key={id}>
                {week.map((day, id) => {
                  if (day === null) return <div />
                  return (
                    <DatePickerDayCell key={id} value={day}>
                      <DatePickerDayCellTrigger>
                        {day.day}
                      </DatePickerDayCellTrigger>
                    </DatePickerDayCell>
                  )
                })}
              </DatePickerRow>
            ))}
          </DatePickerRowGroup>
        </DatePickerGrid>
      )}

      {api.view === 'month' && (
        <DatePickerGrid>
          <DatePickerMonthRowGroup columns={4} format="short">
            {(months) =>
              months.map(([month, value]) => (
                <DatePickerMonthRow key={value}>
                  <DatePickerMonthCell value={value}>
                    <DatePickerMonthCellTrigger>
                      {month}
                    </DatePickerMonthCellTrigger>
                  </DatePickerMonthCell>
                </DatePickerMonthRow>
              ))
            }
          </DatePickerMonthRowGroup>
        </DatePickerGrid>
      )}
    </DatePickerContent>
  )}
</DatePicker>
```
